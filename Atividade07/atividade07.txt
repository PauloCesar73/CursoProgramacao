void bubble_sort (int v[], int n) 
    
int a, b, aux;

for (a = 1; a < n; a++) 
     printf("\n[%d] ", a);

for (b = 0; b < n - 1; b++) 
     printf("%d, ", b);

if (v[b] > v[b + 1]) 
aux  = v[b];
v[b] = v[b + 1];
v[b + 1] = aux;

int i, n=10, v[10] = {4,2,1,5,6,8,9,7,3,10})

for(i=0, i<n; i++)
printif("v[%d]=%d", i, v[i]);

bubble_sort(v,n);

for(i=0, i<n; i++)
printif("v[%d]=%d", i, v[i]);

return 0
